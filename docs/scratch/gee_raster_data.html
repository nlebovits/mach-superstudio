def export_to_drive(image, description, folder, region, scale=250):
    task = ee.batch.Export.image.toDrive(
        image=image,
        description=description,
        folder=folder,
        scale=scale,
        region=region.geometry().bounds().getInfo()['coordinates'],
        fileFormat='GeoTIFF'
    )
    task.start()
